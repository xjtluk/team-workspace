/**
 * 清洗工具输出，剥离所有XML标签和工具调用内容
 * 只保留纯文本回复
 */
function sanitizeReply(text) {
  if (!text) return '';
  
  // 1. 移除 <tool_call>...</tool_call> 及其内容
  text = text.replace(/<tool_call>[\s\S]*?<\/tool_call>/g, '');
  
  // 2. 移除 <function=...>...</function> 及其内容
  text = text.replace(/<function=[^>]*>[\s\S]*?<\/function>/g, '');
  
  // 3. 移除各种工具标签（read_file/write_file/bash等）及内容
  text = text.replace(/<(?:read_file|write_file|bash|list_files|search_code|tool)[^>]*>[\s\S]*?<\/\w+>/g, '');
  
  // 4. 移除自闭合工具标签
  text = text.replace(/<(?:read_file|write_file|bash|list_files|search_code|tool)[^>]*\/>/g, '');
  
  // 5. 移除残留的孤立工具标签
  text = text.replace(/<\/?(?:tool_call|read_file|write_file|bash|list_files|search_code|tool)[^>]*>/g, '');
  
  // 6. 清理多余空行（连续3行以上合并为2行）
  text = text.replace(/\n{3,}/g, '\n\n');
  
  // 7. 首尾去空
  text = text.trim();
  
  return text;
}

// === 新逻辑（强） ===
const RECENT_MSG_WINDOW_MS = 3000;  // 3秒窗口
const recentMsgKeys = new Map();     // key → timestamp

function isDuplicate(author, content, timestamp) {
  // 取内容前50字符作为指纹，忽略微小差异
  const fingerprint = content.substring(0, 50).trim();
  const key = `${author}:${fingerprint}`;
  
  const lastSeen = recentMsgKeys.get(key);
  if (lastSeen && (timestamp - lastSeen) < RECENT_MSG_WINDOW_MS) {
    return true;  // 窗口内重复，丢弃
  }
  
  // 更新时间戳
  recentMsgKeys.set(key, timestamp);
  
  // 定期清理过期条目（防内存泄漏）
  if (recentMsgKeys.size > 1000) {
    const now = Date.now();
    for (const [k, v] of recentMsgKeys) {
      if (now - v > 60000) recentMsgKeys.delete(k);  // 60秒过期
    }
  }
  
  return false;
}
