```javascript
function sanitizeReply(text) {
  if (!text) return text;
  // 剥离 tool_call / read_file / write_file / function= 标签及内容
  text = text.replace(/<tool_call>[\s\S]*?<\/tool_call>/g, '');
  text = text.replace(/<(read_file|write_file|list_files|search_code|bash)[^>]*>[\s\S]*?<\/\1>/g, '');
  text = text.replace(/<function=[^>]*>[\s\S]*?<\/function>/g, '');
  text = text.replace(/<shell_executor>[\s\S]*?<\/shell_executor>/g, '');
  // 清理多余空行
  text = text.replace(/\n{3,}/g, '\n\n').trim();
  return text;
}
```

然后在发送处改成：`const cleanContent = sanitizeReply(content);`，用 `cleanContent` 发送。
